<script>
    let clickConfirmed = false;

    // Funkcja sprawdza, czy użytkownik kliknął w odpowiednim miejscu (na twarz)
    function handleImageClick(event) {
        const img = document.getElementById('captcha-img');
        const rect = img.getBoundingClientRect();
        
        // Obliczanie współrzędnych kliknięcia (w procentach)
        const x = event.clientX - rect.left;
        const y = event.clientY - rect.top;
        
        // Pokaż prostokąt zaznaczenia
        const selection = document.getElementById('selection-box');
        selection.style.left = (x - 25) + 'px';
        selection.style.top = (y - 25) + 'px';
        selection.style.width = '50px';
        selection.style.height = '50px';
        selection.style.display = 'block';

        // Szacunkowy obszar twarzy na pierwszym zdjęciu to około 30-40% szerokości / 60-75% wysokości
        const percentageX = (x / rect.width) * 100;
        const percentageY = (y / rect.height) * 100;

        // Przykład: uznajemy trafienie, jeśli użytkownik kliknie w okolice twarzy (np. w zakresie 35-45% X, 60-70% Y)
        if (percentageX >= 30 && percentageX <= 48 && percentageY >= 58 && percentageY <= 75) {
            clickConfirmed = true;
        } else {
            clickConfirmed = false;
        }
    }

    function verifySelection() {
        if (clickConfirmed) {
            document.getElementById('captcha-view').style.display = 'none';
            document.getElementById('success-view').style.display = 'block';
        } else {
            alert('Niepoprawne zaznaczenie. Spróbuj ponownie.');
            resetSelection();
        }
    }

    function resetSelection() {
        document.getElementById('selection-box').style.display = 'none';
        clickConfirmed = false;
    }

    function resetCaptcha() {
        document.getElementById('success-view').style.display = 'none';
        document.getElementById('captcha-view').style.display = 'block';
        resetSelection();
    }
</script>
